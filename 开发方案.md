# 奥维数据格式转换工具 - 开发方案

## 1. 项目概述

开发一个 Python 桌面应用，支持将奥维地图的 OVOBJ 和 OVKML 格式数据转换为标准 KML、Shapefile (SHP) 和 DXF (CAD) 格式。

### 核心需求
- **输入格式**: OVOBJ (二进制), OVKML (XML)
- **输出格式**: KML, SHP, DXF
- **坐标处理**: 保持原始坐标系不变（不做坐标转换）
- **界面**: PyQt5 图形界面
- **批量处理**: 支持多文件批量转换

## 2. 格式分析结果

### 2.1 OVKML 格式 (已完全解析)

OVKML 是标准 KML 的扩展，XML 结构如下：

```xml
<?xml version="1.0" encoding="UTF-8"?>
<kml xmlns="http://www.opengis.net/kml/2.2">
<Document>
    <name>文档名称</name>
    <Folder>
        <name>文件夹名称</name>
        <Placemark>
            <name>对象名称</name>
            <description>描述信息</description>
            <OvAttr>
                <OvIcon>4</OvIcon>        <!-- 图标类型 -->
                <OvIconNum>0</OvIconNum>  <!-- 图标编号 -->
            </OvAttr>
            <Style>...</Style>
            <OvCoordType>CGCS2000</OvCoordType>  <!-- 坐标系类型 -->
            <Point>
                <coordinates>经度,纬度,高程</coordinates>
            </Point>
        </Placemark>
    </Folder>
</Document>
</kml>
```

**OVKML 特有元素**:
- `<OvAttr>`: 奥维属性（图标、图标编号）
- `<OvCoordType>`: 坐标系类型（CGCS2000/GCJ02/WGS84）
- 坐标顺序: `经度,纬度,高程`（与标准 KML 一致）

### 2.2 OVOBJ 格式 (已逆向解析)

OVOBJ 是二进制格式，结构如下：

```
文件头:
  [0x00-0x04] Magic: "OviO" (4 bytes)
  [0x04-0x0C] 保留 (8 bytes, 全零)
  [0x0C-0x10] 对象总数 (4 bytes, big-endian int)
  [0x10-0xF0] 保留 (大量零填充)

文件夹记录 (重复):
  [0xF0-0xF4] 文件夹名称长度 N (4 bytes, big-endian int)
  [0xF4-0xF4+N] 文件夹名称 (UTF-8 字符串)
  后续字段...

对象记录 (重复):
  [coord+0] 纬度 (8 bytes, little-endian double)
  [coord+8] 经度 (8 bytes, little-endian double)
  [coord+16] 填充 (4 bytes, big-endian int, 通常为 0)
  [coord+20] 对象类型 (4 bytes, big-endian int, 4=点)
  [coord+24] 标志位 (4 bytes, big-endian int, 1=有名称)
  [coord+28] 名称长度 N (4 bytes, big-endian int)
  [coord+32] 名称 (N bytes, UTF-8 字符串)
  [coord+32+N] 属性数据 (变长)
```

**关键发现**:
- 坐标存储顺序: `纬度, 经度`（与 KML 相反）
- 整数使用 big-endian 字节序
- 浮点数使用 little-endian 字节序
- 对象类型 4 = 点对象
- 样例文件包含 32 个点对象，分布在 3 个文件夹中

```python
from dataclasses import dataclass
from typing import List, Optional
from enum import Enum

class CoordType(Enum):
    WGS84 = "WGS84"
    GCJ02 = "GCJ02"
    CGCS2000 = "CGCS2000"

class GeoType(Enum):
    POINT = "Point"
    LINE = "LineString"
    POLYGON = "Polygon"

@dataclass
class GeoPoint:
    lon: float
    lat: float
    alt: float = 0.0

@dataclass
class GeoObject:
    name: str
    description: Optional[str]
    geo_type: GeoType
    coordinates: List[GeoPoint]
    coord_type: CoordType
    style: Optional[dict] = None
    attributes: Optional[dict] = None

@dataclass
class GeoFolder:
    name: str
    objects: List[GeoObject]

@dataclass
class GeoDocument:
    name: str
    folders: List[GeoFolder]
    coord_type: CoordType
```

## 4. 实现细节

### 4.1 OVOBJ 解析器

```python
# parsers/ovobj_parser.py
import struct
from typing import List, Tuple
from models.geo_objects import *

class OvobjParser:
    MAGIC = b'OviO'
    
    def parse(self, filepath: str) -> GeoDocument:
        with open(filepath, 'rb') as f:
            data = f.read()
        
        if data[:4] != self.MAGIC:
            raise ValueError("不是有效的 OVOBJ 文件")
        
        # 解析文件头
        obj_count = struct.unpack('>I', data[0x0C:0x10])[0]
        
        # 解析文件夹和对象
        folders = []
        pos = self._find_first_record(data)
        
        while pos < len(data):
            # 检查是否是文件夹名称
            folder = self._try_parse_folder(data, pos)
            if folder:
                folders.append(folder)
                pos = folder['_end']
            else:
                break
        
        return GeoDocument(
            name=Path(filepath).stem,
            folders=folders,
            coord_type=CoordType.CGCS2000  # 从文件头推断
        )
    
    def _parse_object(self, data: bytes, pos: int) -> Tuple[GeoObject, int]:
        """解析单个对象"""
        lat = struct.unpack('<d', data[pos:pos+8])[0]
        lon = struct.unpack('<d', data[pos+8:pos+16])[0]
        pos += 16
        
        padding = struct.unpack('>I', data[pos:pos+4])[0]
        pos += 4
        
        obj_type = struct.unpack('>I', data[pos:pos+4])[0]
        pos += 4
        
        flag = struct.unpack('>I', data[pos:pos+4])[0]
        pos += 4
        
        name_len = struct.unpack('>I', data[pos:pos+4])[0]
        pos += 4
        
        name = data[pos:pos+name_len].decode('utf-8')
        pos += name_len
        
        # 解析额外属性...
        # TODO: 解析属性数据
        
        geo_type = self._type_from_code(obj_type)
        coords = [GeoPoint(lon=lon, lat=lat)]
        
        return GeoObject(
            name=name,
            description=None,
            geo_type=geo_type,
            coordinates=coords,
            coord_type=CoordType.CGCS2000
        ), pos
```

### 4.2 KML 写入器

```python
# writers/kml_writer.py
import xml.etree.ElementTree as ET
from xml.dom import minidom

class KmlWriter:
    KML_NS = "http://www.opengis.net/kml/2.2"
    
    def write(self, doc: GeoDocument, filepath: str):
        root = ET.Element('kml')
        root.set('xmlns', self.KML_NS)
        
        doc_elem = ET.SubElement(root, 'Document')
        ET.SubElement(doc_elem, 'name').text = doc.name
        
        for folder in doc.folders:
            self._write_folder(doc_elem, folder)
        
        # 格式化输出
        xml_str = minidom.parseString(ET.tostring(root)).toprettyxml()
        with open(filepath, 'w', encoding='utf-8') as f:
            f.write(xml_str)
    
    def _write_folder(self, parent, folder):
        folder_elem = ET.SubElement(parent, 'Folder')
        ET.SubElement(folder_elem, 'name').text = folder.name
        
        for obj in folder.objects:
            self._write_placemark(folder_elem, obj)
    
    def _write_placemark(self, parent, obj):
        pm = ET.SubElement(parent, 'Placemark')
        ET.SubElement(pm, 'name').text = obj.name
        
        if obj.description:
            ET.SubElement(pm, 'description').text = obj.description
        
        # 写入坐标
        if obj.geo_type == GeoType.POINT:
            point = ET.SubElement(pm, 'Point')
            coord = obj.coordinates[0]
            ET.SubElement(point, 'coordinates').text = f"{coord.lon},{coord.lat},{coord.alt}"
        elif obj.geo_type == GeoType.LINE:
            line = ET.SubElement(pm, 'LineString')
            coords_text = ' '.join(f"{c.lon},{c.lat},{c.alt}" for c in obj.coordinates)
            ET.SubElement(line, 'coordinates').text = coords_text
        elif obj.geo_type == GeoType.POLYGON:
            # 类似处理...
            pass
```

### 4.3 SHP 写入器

```python
# writers/shp_writer.py
import shapefile

class ShpWriter:
    def write(self, doc: GeoDocument, filepath: str):
        # 根据几何类型分别写入
        points = []
        lines = []
        polygons = []
        
        for folder in doc.folders:
            for obj in folder.objects:
                if obj.geo_type == GeoType.POINT:
                    points.append(obj)
                elif obj.geo_type == GeoType.LINE:
                    lines.append(obj)
                elif obj.geo_type == GeoType.POLYGON:
                    polygons.append(obj)
        
        if points:
            self._write_points(points, filepath.replace('.shp', '_points.shp'))
        if lines:
            self._write_lines(lines, filepath.replace('.shp', '_lines.shp'))
        if polygons:
            self._write_polygons(polygons, filepath.replace('.shp', '_polygons.shp'))
    
    def _write_points(self, objects, filepath):
        w = shapefile.Writer(filepath)
        w.field('NAME', 'C', 100)
        w.field('DESC', 'C', 254)
        
        for obj in objects:
            coord = obj.coordinates[0]
            w.point(coord.lon, coord.lat)
            w.record(obj.name, obj.description or '')
        
        w.close()
```

### 4.4 DXF 写入器

```python
# writers/dxf_writer.py
import ezdxf

class DxfWriter:
    def write(self, doc: GeoDocument, filepath: str):
        dwg = ezdxf.new('R2010')
        msp = dwg.modelspace()
        
        for folder in doc.folders:
            # 创建图层
            dwg.layers.new(folder.name)
            
            for obj in folder.objects:
                self._write_object(msp, obj, folder.name)
        
        dwg.saveas(filepath)
    
    def _write_object(self, msp, obj, layer_name):
        if obj.geo_type == GeoType.POINT:
            coord = obj.coordinates[0]
            msp.add_point(
                (coord.lon, coord.lat, coord.alt),
                dxfattribs={'layer': layer_name}
            )
        elif obj.geo_type == GeoType.LINE:
            points = [(c.lon, c.lat, c.alt) for c in obj.coordinates]
            msp.add_lwpolyline(
                [(p[0], p[1]) for p in points],
                dxfattribs={'layer': layer_name}
            )
```
